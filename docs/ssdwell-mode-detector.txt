// ============================================================================
//  SSDwellModeDetector — intelligent selection of the dwell operating mode
// ============================================================================
//
//  Decides HOW SimonSays should drive its dwell-click, by fusing several
//  signals of differing reliability:
//
//    (1) User calibration   — two probe buttons, one meant to be activated by
//                             mouse and one by gaze. Ground truth. Authoritative.
//    (2) HID gaze stream     — SSGazeReader producing valid samples.
//    (3) Cursor kinematics    — classifies whether the cursor is being driven
//                             by a hand or by gaze, from its motion signature.
//    (4) Passive detection    — HID device present, known tool process running,
//                             Windows Eye Control enabled (presence, not proof).
//
//  Output: a recommended SSDwellMode plus a confidence value. SSDwellConfig
//  can run in a forced mode (user override) or in Auto mode, in which case it
//  follows this detector.
//
//  This file depends on:
//    - SSGazeReader     (HID reading, from Option B)
//    - EyeTrackingStatus / DetectEyeTracking() (passive detection helpers)
//    - SSDwellConfig    (global config singleton)
// ============================================================================

#pragma once

#include <windows.h>
#include <array>
#include <atomic>
#include <cmath>
#include <cstdint>
#include <mutex>
#include <optional>

// ----------------------------------------------------------------------------
//  The four ways the dwell can operate.
// ----------------------------------------------------------------------------
enum class SSDwellMode {
    Off,            // No dwell. User clicks by other means.
    MouseDwell,     // Dwell follows the Windows cursor (tool moves the cursor,
                    // or a real mouse / head tracker is in use).
    HidDwell,       // Dwell follows gaze read directly from the HID device
                    // (external tool does NOT move the cursor — two-step modes).
    ExternalClick,  // A tool already issues real clicks on our buttons; our own
                    // dwell is unnecessary and stays out of the way.
};

// User-facing selection. Auto delegates to SSDwellModeDetector.
enum class SSDwellModeSelection {
    Auto,
    ForceOff,
    ForceMouseDwell,
    ForceHidDwell,
};

// ============================================================================
//  MouseMotionClassifier
// ----------------------------------------------------------------------------
//  Samples the cursor at a fixed rate and classifies the likely driver of its
//  motion as hand vs. gaze, from kinematic features over a sliding window.
// ============================================================================
class MouseMotionClassifier {
public:
    struct Verdict {
        float gazeLikelihood;   // 0..1 — how "gaze-like" the motion looks
        bool  cursorFrozen;     // true if the cursor barely moved at all
        bool  enoughData;       // false until the window has filled
    };

    void Reset() {
        std::lock_guard<std::mutex> lock(m_mutex);
        m_count = 0;
        m_head  = 0;
    }

    // Call at a fixed cadence (e.g. 60 Hz) with the current cursor position
    // and a millisecond timestamp (GetTickCount).
    void AddSample(POINT cursor, DWORD timestampMs) {
        std::lock_guard<std::mutex> lock(m_mutex);
        Sample& s = m_ring[m_head];
        s.pt = cursor;
        s.t  = timestampMs;
        m_head = (m_head + 1) % kWindow;
        if (m_count < kWindow) ++m_count;
    }

    Verdict Classify() const {
        std::lock_guard<std::mutex> lock(m_mutex);
        Verdict v{ 0.0f, false, false };
        if (m_count < kMinSamples) return v;
        v.enoughData = true;

        // Walk the ring in chronological order.
        const int n = m_count;
        auto at = [&](int i) -> const Sample& {
            int idx = (m_head - m_count + i + kWindow) % kWindow;
            return m_ring[idx];
        };

        int    teleports     = 0;   // implausibly fast single-frame jumps
        int    movingFrames  = 0;   // frames with any motion
        int    stillFrames   = 0;   // frames essentially static
        double sumSpeed      = 0.0;
        double peakSpeed      = 0.0;
        double jitterAccum    = 0.0; // residual motion during slow phases
        int    jitterFrames   = 0;

        for (int i = 1; i < n; ++i) {
            const Sample& a = at(i - 1);
            const Sample& b = at(i);
            const double dt = std::max<DWORD>(1, b.t - a.t) / 1000.0; // seconds
            const double dx = double(b.pt.x - a.pt.x);
            const double dy = double(b.pt.y - a.pt.y);
            const double dist = std::sqrt(dx * dx + dy * dy);
            const double speed = dist / dt; // px/s

            sumSpeed += speed;
            peakSpeed = std::max(peakSpeed, speed);

            if (dist < kStillPx) {
                ++stillFrames;
            } else {
                ++movingFrames;
            }

            // A jump beyond kTeleportPx within a single ~16 ms frame is far
            // too fast for a hand: it is a saccade or a two-step teleport.
            if (dist > kTeleportPx && dt < 0.05) {
                ++teleports;
            }

            // During slow phases, accumulate residual jitter. Hands holding
            // still produce ~0; gaze fixation always shimmers.
            if (speed < kFixationSpeed) {
                jitterAccum += dist;
                ++jitterFrames;
            }
        }

        // Cursor essentially never moved across the whole window.
        if (movingFrames <= n / 20) {
            v.cursorFrozen = true;
        }

        const double avgSpeed   = sumSpeed / std::max(1, n - 1);
        const double fixJitter  = jitterFrames ? jitterAccum / jitterFrames : 0.0;
        const double teleRatio  = double(teleports) / std::max(1, n - 1);

        // --- Heuristic scoring ------------------------------------------------
        // Each cue contributes evidence toward "gaze-driven". Tuned for typical
        // 30–120 Hz trackers; expose as config if you want to fine-tune.
        float score = 0.0f;

        // Saccade-like teleports are the strongest single cue.
        if (teleRatio > 0.02) score += 0.40f;
        else if (teleRatio > 0.005) score += 0.20f;

        // Persistent fixation jitter (hand-still would be ~0).
        if (fixJitter > kJitterHigh) score += 0.30f;
        else if (fixJitter > kJitterLow) score += 0.15f;

        // Bimodal velocity: very high peak relative to a low average.
        if (peakSpeed > 4000.0 && avgSpeed < 400.0) score += 0.30f;
        else if (peakSpeed > 2500.0 && avgSpeed < 600.0) score += 0.15f;

        v.gazeLikelihood = std::min(1.0f, score);
        return v;
    }

private:
    struct Sample { POINT pt; DWORD t; };

    static constexpr int    kWindow        = 120;   // ~2 s at 60 Hz
    static constexpr int    kMinSamples    = 45;    // need ~0.75 s before judging
    static constexpr double kStillPx       = 1.5;   // below this = "still" frame
    static constexpr double kTeleportPx    = 150.0; // single-frame jump = saccade
    static constexpr double kFixationSpeed = 300.0; // px/s threshold for fixation
    static constexpr double kJitterLow     = 0.8;   // px residual per slow frame
    static constexpr double kJitterHigh    = 2.0;

    mutable std::mutex          m_mutex;
    std::array<Sample, kWindow> m_ring{};
    int                         m_head  = 0;
    int                         m_count = 0;
};

// ============================================================================
//  Calibration probe — the two-button ground-truth mechanism
// ----------------------------------------------------------------------------
//  SimonSays shows (e.g. in a setup screen) two probe buttons:
//      * "Activate me with the MOUSE"  -> records CalibrationResult::Mouse
//      * "Activate me with your GAZE"   -> records CalibrationResult::Gaze
//
//  How each is activated is inferred from what was happening at activation:
//      - If our HID reader had a valid gaze sample inside the button at the
//        moment of activation -> gaze path confirmed.
//      - If the cursor sat inside the button and motion looked gaze-driven
//        -> the external tool drives the cursor by gaze (MouseDwell works).
//      - If motion looked hand-driven -> plain mouse.
//
//  The user pressing the GAZE button successfully is the single most reliable
//  signal we can obtain, because it proves the whole pipeline end to end.
// ============================================================================
enum class CalibrationOutcome {
    None,
    ConfirmedHidGaze,        // gaze button hit, HID stream was valid -> HidDwell
    ConfirmedCursorIsGaze,   // gaze button hit via cursor -> MouseDwell (tool moves cursor)
    ConfirmedMouseIsHand,    // mouse button hit with hand-like motion -> mouse user
};

// ============================================================================
//  SSDwellModeDetector
// ============================================================================
class SSDwellModeDetector {
public:
    static SSDwellModeDetector& Instance() {
        static SSDwellModeDetector inst;
        return inst;
    }

    // Drive the classifier. Call from a fixed-rate timer (~60 Hz).
    void Tick() {
        POINT pt;
        if (GetCursorPos(&pt)) {
            m_motion.AddSample(pt, GetTickCount());
        }
    }

    // Feed calibration outcomes from the probe buttons.
    void ReportCalibration(CalibrationOutcome outcome) {
        std::lock_guard<std::mutex> lock(m_mutex);
        m_calibration = outcome;
        m_calibrationTick = GetTickCount();
    }

    // Passive detection result (presence of devices / tools / Eye Control).
    // Pass in the struct produced by DetectEyeTracking(); kept abstract here.
    struct PassiveSignals {
        bool hidDevicePresent   = false;
        bool externalToolRunning = false;
        bool windowsEyeControl  = false;
        bool toolKnownToClick   = false; // tool issues real clicks (TD Control dwell, WEC left-click)
    };

    void UpdatePassive(const PassiveSignals& s) {
        std::lock_guard<std::mutex> lock(m_mutex);
        m_passive = s;
    }

    // Whether the HID reader is currently delivering valid gaze.
    void SetHidLive(bool live) { m_hidLive.store(live); }

    // ----- The decision --------------------------------------------------------
    struct Decision {
        SSDwellMode mode;
        float       confidence;   // 0..1
        const wchar_t* reason;    // for status UI / logging
    };

    Decision Decide() const {
        std::lock_guard<std::mutex> lock(m_mutex);

        // 1) Calibration is authoritative while it is fresh (e.g. < 10 min).
        const DWORD now = GetTickCount();
        const bool calibFresh =
            (m_calibration != CalibrationOutcome::None) &&
            (now - m_calibrationTick < kCalibrationValidMs);

        if (calibFresh) {
            switch (m_calibration) {
            case CalibrationOutcome::ConfirmedHidGaze:
                return { SSDwellMode::HidDwell, 1.0f, L"Calibrated: HID gaze" };
            case CalibrationOutcome::ConfirmedCursorIsGaze:
                return { SSDwellMode::MouseDwell, 1.0f, L"Calibrated: gaze-driven cursor" };
            case CalibrationOutcome::ConfirmedMouseIsHand:
                return { SSDwellMode::MouseDwell, 0.9f, L"Calibrated: manual mouse" };
            default: break;
            }
        }

        // 2) A tool that already clicks on our buttons -> stay out of the way.
        if (m_passive.toolKnownToClick) {
            return { SSDwellMode::ExternalClick, 0.8f, L"External tool issues clicks" };
        }

        // 3) Live HID gaze available -> prefer reading it directly.
        const MouseMotionClassifier::Verdict v = m_motion.Classify();
        if (m_hidLive.load()) {
            // If the cursor is also moving in a gaze-like way, the tool is
            // probably driving the cursor too; MouseDwell is simpler and works.
            if (v.enoughData && !v.cursorFrozen && v.gazeLikelihood > 0.6f) {
                return { SSDwellMode::MouseDwell, 0.7f, L"HID + gaze-driven cursor" };
            }
            // Cursor frozen but HID flowing = classic two-step mode -> HID dwell.
            return { SSDwellMode::HidDwell, 0.85f, L"HID gaze, cursor static" };
        }

        // 4) No HID stream. Lean on cursor kinematics.
        if (v.enoughData) {
            if (v.cursorFrozen && m_passive.AnyPresence()) {
                // Something is present but the cursor isn't moving and we can't
                // read HID: dwell can't track. Recommend Off and let the UI
                // prompt the user to enable a cursor mode or run calibration.
                return { SSDwellMode::Off, 0.4f, L"Tracker present, cursor static, no HID" };
            }
            if (v.gazeLikelihood > 0.6f) {
                return { SSDwellMode::MouseDwell, 0.7f, L"Cursor looks gaze-driven" };
            }
            if (v.gazeLikelihood < 0.25f && !m_passive.AnyPresence()) {
                // Clearly a hand, nothing detected: dwell off by default so a
                // mouse user isn't surprised by auto-firing buttons.
                return { SSDwellMode::Off, 0.6f, L"Manual mouse, no tracker" };
            }
            // Ambiguous middle ground.
            return { SSDwellMode::MouseDwell, 0.4f, L"Ambiguous; defaulting to mouse dwell" };
        }

        // 5) Not enough data yet.
        if (m_passive.hidDevicePresent || m_passive.externalToolRunning ||
            m_passive.windowsEyeControl) {
            return { SSDwellMode::MouseDwell, 0.3f, L"Tracker detected; warming up" };
        }
        return { SSDwellMode::Off, 0.3f, L"No signals yet" };
    }

    void ResetMotion() { m_motion.Reset(); }

private:
    SSDwellModeDetector() = default;
    SSDwellModeDetector(const SSDwellModeDetector&)            = delete;
    SSDwellModeDetector& operator=(const SSDwellModeDetector&) = delete;

    static constexpr DWORD kCalibrationValidMs = 10 * 60 * 1000; // 10 minutes

    mutable std::mutex      m_mutex;
    MouseMotionClassifier   m_motion;
    PassiveSignals          m_passive;
    CalibrationOutcome      m_calibration     = CalibrationOutcome::None;
    DWORD                   m_calibrationTick = 0;
    std::atomic<bool>       m_hidLive{ false };

public:
    // Small helper so PassiveSignals can answer "anything at all?".
    // (Defined out-of-line style for clarity.)
};

// Convenience method on PassiveSignals (declared here to keep the struct above
// readable). In real code, fold this into the struct definition.
inline bool SSDwellModeDetector::PassiveSignals::AnyPresence() const {
    return hidDevicePresent || externalToolRunning || windowsEyeControl;
}

// ============================================================================
//  Wiring sketch
// ----------------------------------------------------------------------------
//  Timer (~60 Hz):
//      SSDwellModeDetector::Instance().Tick();
//
//  Slower timer (~1–3 s) or on WM_DEVICECHANGE:
//      auto st = DetectEyeTracking();
//      SSDwellModeDetector::PassiveSignals ps;
//      ps.hidDevicePresent    = st.hidDevicePresent;
//      ps.externalToolRunning = st.externalToolRunning;
//      ps.windowsEyeControl   = st.windowsEyeControl;
//      ps.toolKnownToClick    = IsClickingTool(st.activeTool); // your mapping
//      SSDwellModeDetector::Instance().UpdatePassive(ps);
//      SSDwellModeDetector::Instance().SetHidLive(SSGazeReader::Instance().HasLiveGaze());
//
//  Apply the decision (only when the user left selection on Auto):
//      if (cfg.GetModeSelection() == SSDwellModeSelection::Auto) {
//          auto d = SSDwellModeDetector::Instance().Decide();
//          cfg.ApplyDetectedMode(d.mode);          // maps mode -> enabled + source
//          UpdateStatusBar(d.reason, d.confidence);
//      }
//
//  Hysteresis: only switch mode if a new decision holds for N consecutive
//  evaluations (e.g. 3) to avoid flapping between modes on noisy data.
// ============================================================================
